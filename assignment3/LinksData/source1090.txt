			$tweetCon.empty();

			if(typeof tweets[type][date] != 'undefined'){
				twcApp.fillContainer(type, date);

				count = 0,
				dateShown = moment(date, 'YYYY-MM-DD').format('MMMM Do, YYYY');

			if ( type == 'retweets' || type == 'favorites' ) {
				_.each(tweets[type][date], function ( tweet ) {
					if ( tweet.by ) {
						count += tweet.by.length;
					}
				});
			}
			else {
				count = tweets[type][date].length;
			}

			switch ( type ) {
				case 'mentions':
					title = (count == 1) ? 'mention' : 'mentions';
					break;
				case 'retweets':
					title = (count == 1) ? 'retweet' : 'retweets';
					relatedTitle = 'Retweeted by';
					break;
				case 'favorites':
					title = (count == 1) ? 'like' : 'likes';
					relatedTitle = 'Liked by';
					break;
			}

			title = count + ' ' + title;

			var listData = {
				title       : title,
				date        : dateShown,
				relatedTitle: relatedTitle,
				tweets      : tweets[type][date]
			};

			$tweetCon.empty().append( tpl(listData) );
		};
