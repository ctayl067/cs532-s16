<script type="text/javascript">
$(document).ready(function() {
	var msg = '#message';
	console.log('test');
	
	$('#likecontent').hide();	

	$('.time').timeago();
	$(msg).autosize();

	$('#post_comment').click(function() {
		$(msg).focus();
	});

	$(msg).keypress(function(e) {
		
		if(e.which == 13) {
			var val = $(msg).val();
			
			$.ajax({
				url: 'pcomments/ajax.php',
				type: 'GET',
				data: 'token=&ptid=&pid=&mid=&type=cmt&msg='+escape(val),
				success: function(data) {
					$(msg).val('');
					$(msg).css('height','14px');
					$('#commentscontainer').append(data);
					$('.time').timeago();
				}
			});
		}
	});

	$('#like_post').click(function() {
		var count = parseFloat($('#count').html()) + 1;
		if(count > 1) {
			$('#if_like').html('You and');
			$('#people').html('others');
		} else {
			$('#if_like').html('You like this.');
			$('#likecontent').hide();
		}
		
		$.ajax({
			url: 'pcomments/ajax.php',
			type: 'GET',
			data: 'token=&ptid=&pid=&mid=&type=like&val=like',
			success: function(data) {
				$(msg).val('');
				$(msg).css('height','14px');
				$('#commentscontainer').append(data);
				$('.time').timeago();
			}
		});		

		$('#like_post').hide();
		$('#unlike_post').show();
	});

	$('#unlike_post').click(function() {
		var count = parseFloat($('#count').html()) - 1;
		if(count < 1) {
			$('#likecontent').hide();	
		}

		$.ajax({
			url: 'pcomments/ajax.php',
			type: 'GET',
			data: 'token=&ptid=&pid=&mid=&type=like&val=unlike',
			success: function(data) {
				$(msg).val('');
				$(msg).css('height','14px');
				$('#commentscontainer').append(data);
				$('.time').timeago();
			}
		});				
		
		$('#unlike_post').hide();
		$('#like_post').show();
		$('#if_like').html('');
		$('#people').html('people');
	});
});
</script>
