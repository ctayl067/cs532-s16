// has to be put at the top. Called before the js file is loaded.
var brokenImages = function() { // object that stores all broken images on the page
	var images = new Array();
	function add(img, name){
		if (!$('SchmapKit.Maps__GStatic.Map.0.container')) {
			var tmp = function tmp () {brokenImages.add(img, name);};
			setTimeout(tmp, 200);
			return;
		}
		else {
			img.src = "/images/spacer.png";
			img.className = "defaultTwitterPic_noBG ";
			images.push([name, img]);
			var url = "http://twitter.com/users/show/"+name+".json?callback=setTwitterImg";
			addScript(url);
			setTimeout(brokenImages.fail.bind(null, name, img), 5000);
		}
	}
	function resolve(name, url){
		// log(name+" "+url);
		for (i=0; i<images.length; i++) {
			if (images[i][0] && images[i][0].toLowerCase()==name.toLowerCase()) {
				images[i][1].src = url;
				images[i][1].className = '';
				images[i][1].onerror = function(){setTwitterImgError(this);};
				images[i][1].onload = function(){checkImage(this, name, true)};
				//images[i][0] = null;
	}}}
	function fail(name, img) {
		for (i=0; i<images.length; i++) {
			if (images[i][0] && images[i][0].toLowerCase()==name.toLowerCase()) {
				if (images[i][1].src.match("/images/spacer.png")) {
					images[i][1].className = "defaultTwitterPic";
					images[i][0] = 'setToDefault';
	}}}}
	return {add:add, resolve:resolve, fail:fail};
}();
function setTwitterImg(r){
	brokenImages.resolve(r.screen_name, replaceMini(r.profile_image_url));
}
function replaceMini(url){
	url1 = url.split('_normal.');
	if (url1[1])
		return url1[0]+"_mini."+url1[1];
	url2 = url.split('_bigger.');
	if (url2[1])
		return url2[0]+"_mini."+url2[1];
	else
		return url;
}
function setTwitterImgError(r) {
	img.src = "/images/spacer.png";
	img.className = "defaultTwitterPic";
	log('broken image not handled');
	return;
}
function checkImage(img, name, secondTry) {
	image = new Image(); image.src = img.src;
	image.onload = function(){
		if (!(this.width > 1 || this.height > 1)){
			if (secondTry) {
				log('broken image not handled');
				img.src = "/images/spacer.png";
				img.className = "defaultTwitterPic";
			}
			else
				brokenImages.add(img, name);
		}
	};
	img.onload = "";
	img.onerror = "";
}
