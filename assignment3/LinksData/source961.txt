	function createCard(top, left, data) {
		var bgpos = 75 * data.pos / 230;
		$("body").append("<div class='userMicroCard' style='top: "+top+"px; left: "+left+"px;'><div class='background' style='background-image: url("+data.bg+"); background-position-y: -"+bgpos+"px'></div><div class='avatar'><img src='"+data.photo+"'></div><div class='username'>"+data.name+"</div><div class='userstatus'>"+data.status+"</div></div>");
		if (data.follow == 1) {
			$(".userMicroCard").append("<div class='subscribe' data-user='"+data.id+"' data-action='follow'>Читать</div>");
		} else if (data.follow == 2) {
			$(".userMicroCard").append("<div class='subscribe unsubs' data-user='"+data.id+"' data-action='unfollow'>Читаю</div>");
		}
	}
	$(".userRoundPhoto,.wallauthor.fl_l").bind("mouseover", function() {
		$(".userMicroCard").remove();
		var target = $(this).attr("data-target");
		var top = $(this).offset().top + $(this).height() + 8;
		var left = $(this).offset().left - (300)/2;
		if (users[target]) {
			createCard(top, left, users[target]);
		} else {
			$.ajax({url : "/lenta/?act=userCard&id="+target, dataType: "json", success: function(data) {
				createCard(top, left, data);
				users[data.id] = data;
			}});
		}
	});
	$(".userMicroCard").live("mouseleave", function() {
		$(".userMicroCard").remove();

	$(".userMicroCard .subscribe").live("click", function() {
		var action = $(this).attr("data-action");
		var id = $(this).attr("data-user");
		var target = $(this);
		if (action == "follow") {
			$.post('/index.php?go=subscriptions&act=add', {for_user_id: id}, function(){
				target.attr({"data-action" : "unfollow"});
				target.addClass("unsubs");
				target.text("Читаю");
			});
		} else if (action == "unfollow") {
			$.post('/index.php?go=subscriptions&act=del', {del_user_id: id}, function(){
				target.attr({"data-action" : "follow"});
				target.removeClass("unsubs");
				target.text("Читать");
			});
		}
	});
